## Tests Structure

```bash
/// <reference types="cypress" />

describe('First test suite', () => {

    describe('suite section', () => {

        beforeEach('login', () => {
            //repeat for every test
        })

        it('First test', () => {
            //put the code of the test
        })

        it('Second test', () => {   
            //put the code of the test
        })
    })

    it('First test', () => {
        //put the code of the test
    })

    it('Second test', () => {
        //put the code of the test
    })

    it('Third test', () => {
        //put the code of the test
    })
})

describe('Second test suite', () => {

    it('First test', () => {
        //put the code of the test
    })

    it('Second test', () => {
        //put the code of the test
    })

    it('Third test', () => {
        //put the code of the test
    })
})
```
